Physical schema (with constraints)

1.  Passanger
    ---------
    PassengerID  INT PK
    Gender VARCHAR(6) NOT NULL
    CustomerType VARCHAR(25) NULL
    Age INT NOT NULL
    Travel_type VARCHAR(25) NULL
    Satisfaction VARCHAR(35) NULL
    FlightID FK >-< Flight.FlightID
    BookingID  FK -< OnlineServices.BookingID
    InflightServiceID FK >- InflightServices.InflightServiceID
    AirlineServiceID FK >-AirlineServices.ArilineServiceID
    
    
2.  Flights
    ------
    FlightID SERIAL PK
    Class VARCHAR(25) NOT NULL
    SeatComfort INT NULL
    FlightDistance INT NOT NULL
    Cleanliness INT NULL
    DepartArriveID FK >- DepartureArrival.DepartArriveID
    
    
3.  DepartureArrival
    -----------------
    DepartArriveID SERIAL PK
    TimeConvenience INT NULL
    DepartureDelay INT NULL
    ArrivalDelay INT NULL
    
    
4.  InflightServices
    -----------------
    InflightServiceID  SERIAL PK
    Wifi INT NULL
    Entertainment INT NULL
    InflightService INT NULL
    FoodDrink INT NULL
    OnboardService INT NULL
    LegroomService INT NULL
    
    
5.  OnlineServices
    ---------------
    BookingID SERIAL PK
    EaseOfOnlineBooking INT NULL
    OnlineBoarding INT NULL
    
    
6.  AirlineServices
    ----------------
    AirlineServiceID SERIAL PK
    CheckinService INT NULL
    GateLocation INT NULL
    BaggageHandling INT NULL
